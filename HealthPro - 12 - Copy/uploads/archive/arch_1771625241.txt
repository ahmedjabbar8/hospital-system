from gpio import *
from time import *

# Define Pins
LDR_PIN = A0      # Analog pin for LDR
LIGHT_PIN = 1     # Digital pin for Light/LED

# Threshold for darkness
THRESHOLD = 500

def setup():
    print("System Starting...")
    pinMode(LIGHT_PIN, OUT)

def loop():
    while True:
        # Read light level (0-1023)
        val = analogRead(LDR_PIN)
        print("Light Level:", val)
        
        # Control Logic
        if val < THRESHOLD:
            digitalWrite(LIGHT_PIN, HIGH)
            print("Status: Dark - Light ON")
        else:
            digitalWrite(LIGHT_PIN, LOW)
            print("Status: Bright - Light OFF")
            
        delay(1000) # Wait 1 second

if __name__ == "__main__":
    setup()
    loop()
